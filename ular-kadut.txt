        function setCookie(name, value, hour) {
            var expires = "";
            if (hour) {
                var date = new Date();
                date.setTime(date.getTime() + (hour * 60 * 60 * 1000));
                expires = "; expires=" + date.toUTCString();
            }
            document.cookie = name + "=" + (value || "") + expires + "; ;domain=.kompas.com;path=/";
        }
